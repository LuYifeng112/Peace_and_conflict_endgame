#Government support 
init_government_support = {
	set_country_flag = { flag = government_support value = 50 }

	if = {
		limit = {
			or = {
				tag = CHI
			}
		}
		modify_country_flag = { flag = government_support value = -7 }
	}

	if = {
		limit = {
			or = {
				tag = PRC
			}
		}
		modify_country_flag = { flag = government_support value = 30 }
	}
}
